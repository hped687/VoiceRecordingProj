let gumStream, recorder; let participantId = "anon"; let questionId = "unknown_question"; let recordingPrompt = ""; let hasSentStartTimer = false; const recordBtn = document.getElementById("record-btn"); const statusMsg = document.getElementById("status-msg"); const recordingIndicator = document.getElementById("recording-indicator"); function showRecordingIndicator() { recordingIndicator.style.display = "flex"; } function hideRecordingIndicator() { recordingIndicator.style.display = "none"; } // Tell parent iframe is ready window.parent.postMessage({ type: "iframe-ready" }, "*"); window.addEventListener("message", (event) => { if (typeof event.data !== "object") return; const { pid, qid, prompt, type } = event.data; if (type === "iframe-ready") return; if (pid && qid) { participantId = pid; questionId = qid; if (prompt) { recordingPrompt = prompt; } } if (type === "stop-recording") { try { if (recorder && gumStream) { recorder.finishRecording(); gumStream.getTracks().forEach(track => track.stop()); recordBtn.innerText = "Recording Finished"; hideRecordingIndicator(); // Tell parent timer can stop window.parent.postMessage({ type: "stop-timer" }, "*"); } } catch (err) { statusMsg.innerText = "‚ö†Ô∏è Error during auto-stop."; console.error("Auto-stop error:", err); } } }); recordBtn.onclick = async function () { if (recordBtn.innerText === "Start Recording") { try { const stream = await navigator.mediaDevices.getUserMedia({ audio: true }); gumStream = stream; const input = new AudioContext().createMediaStreamSource(stream); recorder = new WebAudioRecorder(input, { workerDir: "lib/", encoding: "wav", numChannels: 1, onEncoderLoading: () => (statusMsg.innerText = "Loading encoder..."), onEncoderLoaded: () => (statusMsg.innerText = "Encoder ready. Recording..."), }); recorder.setOptions({ timeLimit: 1800, encodeAfterRecord: true, }); recorder.onComplete = function (rec, blob) { statusMsg.innerText = "Recording complete. Uploading..."; const timestamp = new Date().toISOString().replace(/[:.]/g, "-"); const filename = `${participantId}_${questionId}_${timestamp}.wav`; const formData = new FormData(); formData.append("file", blob, filename); fetch("https://qualtrics-audio-upload.onrender.com/upload", { method: "POST", body: formData, }) .then(() => { statusMsg.innerText = "‚úÖ Uploaded successfully!"; recordBtn.disabled = true; hideRecordingIndicator(); }) .catch((error) => { statusMsg.innerText = "‚ùå Upload failed."; console.error("Upload error:", error); hideRecordingIndicator(); }); }; recorder.startRecording(); recordBtn.innerText = "Stop Recording"; showRecordingIndicator(); if (!hasSentStartTimer) { console.log("üëâ Sending start-timer to parent"); window.parent.postMessage({ type: "start-timer" }, "*"); hasSentStartTimer = true; } } catch (err) { statusMsg.innerText = "‚ö†Ô∏è Microphone access denied."; console.error("Recording error:", err); } } else { try { recorder.finishRecording(); gumStream.getTracks().forEach(track => track.stop()); recordBtn.innerText = "Recording Finished"; hideRecordingIndicator(); window.parent.postMessage({ type: "stop-timer" }, "*"); hasSentStartTimer = false; } catch (err) { statusMsg.innerText = "‚ö†Ô∏è Error stopping recording."; console.error(err); } } };
